## Q01: 选股策略

### 一、多因子选股模型

#### 1. 因子分类

| 因子类型 | 具体因子 | 说明 |
|---------|---------|------|
| **价值因子** | PE、PB、PS、PCF | 估值越低越好 |
| **成长因子** | 营收增速、利润增速 | 增速越高越好 |
| **质量因子** | ROE、毛利率、净利率 | 越高越好 |
| **动量因子** | 20 日涨幅、60 日涨幅 | 强者恒强 |
| **规模因子** | 总市值、流通市值 | 小盘效应 |
| **资金因子** | 主力净流入、北向持股 | 资金青睐 |

---

#### 2. 因子评分法

```python
def 多因子选股 (股票池):
    结果 = []
    
    for 股票 in 股票池:
        # 获取数据
        PE = 股票.市盈率
        PB = 股票.市净率
        ROE = 股票.ROE
        营收增速 = 股票.营收同比增长
        利润增速 = 股票.净利润同比增长
        20 日涨幅 = 股票.涨幅 (20)
        主力净流入 = 股票.主力净流入占比
        
        # 因子评分 (0-10 分)
        PE 得分 = max(0, min(10, (50 - PE) / 5))  # PE 越低分越高
        PB 得分 = max(0, min(10, (10 - PB) / 1))  # PB 越低分越高
        ROE 得分 = max(0, min(10, ROE / 2))  # ROE 越高分越高
        成长得分 = max(0, min(10, (营收增速 + 利润增速) / 4))
        动量得分 = max(0, min(10, 20 日涨幅 / 2))
        资金得分 = max(0, min(10, 主力净流入 / 2))
        
        # 加权总分
        总分 = (
            PE 得分 * 0.20 +      # 价值 20%
            PB 得分 * 0.15 +      # 价值 15%
            ROE 得分 * 0.20 +     # 质量 20%
            成长得分 * 0.20 +     # 成长 20%
            动量得分 * 0.15 +     # 动量 15%
            资金得分 * 0.10       # 资金 10%
        )
        
        结果.append({
            '股票': 股票.代码，
            '名称': 股票.名称，
            '总分': 总分,
            'PE': PE,
            'ROE': ROE,
            '成长': 成长得分，
            '动量': 动量得分
        })
    
    # 按总分排序
    结果.sort(key=lambda x: x['总分'], reverse=True)
    
    return 结果 [:10]  # 返回前 10 名
```

```python
def 动量因子计算 (股票，周期):
    """
    计算动量因子
    """
    当前价格 = 股票.当前价
    N 日前价格 = 股票.N 日前收盘价
    
    动量 = (当前价格 - N 日前价格) / N 日前价格 * 100
    
    return 动量
```

#### 2. 动量选股策略

```python
def 动量选股 (股票池):
    """
    动量策略选股
    """
    结果 = []
    
    for 股票 in 股票池:
        # 计算多周期动量
        M5 = 动量因子计算 (股票，5)
        M20 = 动量因子计算 (股票，20)

        MA10 = 股票.MA10
        MA20 = 股票.MA20
        MA60 = 股票.MA60

        动量得分 = M5 * 0.3 + M20 * 0.4 + M60 * 0.3
        趋势得分 = 10 if 多头排列 else 0
        
        总分 = 动量得分 + 趋势得分
        
        # 过滤条件
        if M20 > 0 and M60 > 0:  # 中线上行

                '股票': 股票.代码，
                '名称': 股票.名称，
                '总分': 总分,
                'M5': M5,
                'M20': M20,
                'M60': M60,
                '多头': 多头排列
            })
    
    # 排序
    结果.sort(key=lambda x: x['总分'], reverse=True)
    
    return 结果 [:10]
```

### 三、价值策略

#### 1. 价值因子

```python
def 价值评分 (股票):
    """
    价值投资评分
    """
    PE = 股票.市盈率
    PB = 股票.市净率
    PS = 股票.市销率
    PCF = 股票.市现率
    股息率 = 股票.股息率
    
    # 行业对比 (百分位)
    PE 百分位 = 行业 PE 百分位 (PE)
    PB 百分位 = 行业 PB 百分位 (PB)
    
    # 评分
    估值得分 = (100 - PE 百分位) / 10  # 百分位越低越好
    股息得分 = min(10, 股息率 * 2)  # 股息率越高分越高
    
    # 质量因子
    ROE = 股票.ROE
    毛利率 = 股票.毛利率
    负债率 = 股票.资产负债率
    
    质量得分 = (
        ROE / 2 +                    # ROE 评分
        毛利率 / 5 +                  # 毛利率评分
        max(0, (70 - 负债率) / 7)     # 负债率评分
    ) / 3
    
    # 成长因子
    营收增速 = 股票.营收同比增长
    利润增速 = 股票.净利润同比增长
    
    成长得分 = (营收增速 + 利润增速) / 4
    
    # 综合得分
    总分 = (
        估值得分 * 0.30 +
        股息得分 * 0.20 +
        质量得分 * 0.30 +
        成长得分 * 0.20
    )
    
    return {
        '股票': 股票.代码，
        '总分': 总分,
        '估值': 估值得分，
        '股息': 股息得分，
        '质量': 质量得分，
        '成长': 成长得分
    }
```

#### 2. 价值选股条件

```python
def 价值选股 (股票池):
    """
    价值投资选股条件
    """
    结果 = []
    
    for 股票 in 股票池:
        # 基本条件
        if 股票.PE < 0:  # 排除亏损
            continue
        if 股票.PE > 50:  # 排除高估值
            continue
        if 股票.ROE < 10:  # ROE 低于 10%
            continue
        if 股票.资产负债率 > 70:  # 负债过高
            continue
        if 股票.上市天数 < 365:  # 上市不足 1 年
            continue
        
        # 价值评分
        评分 = 价值评分 (股票)
        
        if 评分 ['总分'] >= 7:  # 总分 7 分以上
            结果.append (评分)
    
    return 结果
```

## Q02: 择时策略

### 一、趋势跟踪策略

#### 1. 双均线策略

```python
class 双均线策略:
    def __init__(self, 快周期=5, 慢周期=20):
        self.快周期 = 快周期
        self.慢周期 = 慢周期
        self.持仓 = False
        self.买入价 = 0
    
    def 信号 (self, 数据):
        """
        生成交易信号
        """
        MA 快 = 数据.MA(self.快周期)
        MA 慢 = 数据.MA(self.慢周期)
        当前价 = 数据.当前价
        
        # 金叉买入
        if not self.持仓 and MA 快 > MA 慢 and 数据.前 MA 快 <= 数据.前 MA 慢:
            self.持仓 = True
            self.买入价 = 当前价
            return '买入'
        
        # 死叉卖出
        if self.持仓 and MA 快 < MA 慢 and 数据.前 MA 快 >= 数据.前 MA 慢:
            self.持仓 = False
            收益率 = (当前价 - self.买入价) / self.买入价
            return '卖出'
        
        return '持有' if self.持仓 else '空仓'
```

### 二、均值回归策略

#### 1. 布林带均值回归

```python
class 布林带策略:
    def __init__(self, 周期=20, 倍数=2):
        self.周期 = 周期
        self.倍数 = 倍数
        self.持仓 = False
        self.买入价 = 0
    
    def 信号 (self, 数据):
        当前价 = 数据.当前价
        中轨 = 数据.MA(self.周期)
        标准差 = 数据.标准差 (self.周期)
        上轨 = 中轨 + self.倍数 * 标准差
        下轨 = 中轨 - self.倍数 * 标准差
        
        # 触及下轨买入 (均值回归)
        if not self.持仓 and 当前价 < 下轨:
            self.持仓 = True
            self.买入价 = 当前价
            return '买入'
        
        # 回归中轨卖出
        if self.持仓 and 当前价 > 中轨:
            self.持仓 = False
            收益率 = (当前价 - self.买入价) / self.买入价
            return '卖出'
        
        # 止损：跌破下轨 5%
        if self.持仓 and 当前价 < self.买入价 * 0.95:
            self.持仓 = False
            return '止损卖出'
        
        return '持有' if self.持仓 else '空仓'
```

## Q03: 回测方法

### 一、回测框架

#### 1. 回测类结构

```python
class 回测框架:
    def __init__(self, 初始资金=1000000):
        self.初始资金 = 初始资金
        self.当前资金 = 初始资金
        self.持仓 = {}
        self.交易记录 = []
        self.资金曲线 = []
    
    def 运行回测 (self, 策略，数据，开始日期，结束日期):
        """
        运行回测
        """
        for 日期 in 数据.日期范围 (开始日期，结束日期):
            当日数据 = 数据.get(日期)
            
            # 获取策略信号
            信号 = 策略.信号 (当日数据)
            
            # 执行交易
            if 信号 == '买入':
                self.执行买入 (当日数据)
            elif 信号 == '卖出':
                self.执行卖出 (当日数据)
            
            # 更新资金曲线
            总资产 = self.计算总资产 (当日数据)
            self.资金曲线.append({
                '日期': 日期，
                '总资产': 总资产，
                '收益率': (总资产 - self.初始资金) / self.初始资金
            })
        
        return self.生成报告 ()
```

#### 2. 交易执行

```python
def 执行买入 (self, 数据):
    """
    执行买入操作
    """
    价格 = 数据.当前价
    
    # 计算可买数量 (95% 仓位，留 5% 现金)
    可用资金 = self.当前资金 * 0.95
    数量 = int(可用资金 / 价格 / 100) * 100  # 100 股整数倍
    
    if 数量 > 0:
        金额 = 数量 * 价格
        手续费 = 金额 * 0.0003  # 万分之三
        
        self.当前资金 -= (金额 + 手续费)
        self.持仓 [数据.代码] = {
            '数量': 数量，
            '成本': 价格，
            '市值': 金额
        }
        
        self.交易记录.append({
            '日期': 数据.日期，
            '代码': 数据.代码，
            '操作': '买入',
            '价格': 价格，
            '数量': 数量，
            '金额': 金额
        })
```

#### 2. 风险指标

```python
def 计算风险指标 (self):
    """
    计算风险相关指标
    """
    收益率序列 = [d['收益率'] for d in self.资金曲线]
    
    # 最大回撤
    最高值 = 0
    最大回撤 = 0
    for 收益 in 收益率序列:
        总资产 = self.初始资金 * (1 + 收益)
        if 总资产 > 最高值:
            最高值 = 总资产
        回撤 = (最高值 - 总资产) / 最高值
        if 回撤 > 最大回撤:
            最大回撤 = 回撤
    
    # 波动率 (年化)
    日收益率 = np.diff(收益率序列)
    波动率 = np.std(日收益率) * np.sqrt(252)
    
    # 夏普比率
    无风险利率 = 0.03  # 假设 3%
    夏普比率 = (np.mean(日收益率) * 252 - 无风险利率) / 波动率
    
    return {
        '最大回撤': 最大回撤，
        '波动率': 波动率，
        '夏普比率': 夏普比率
    }
```

#### 3. 交易指标

```python
def 计算交易指标 (self):
    """
    计算交易相关指标
    """
    交易次数 = len(self.交易记录) // 2  # 买入卖出算一次
    
    # 胜率
    盈利交易 = 0
    总盈亏 = 0
    for i in range(0, len(self.交易记录), 2):
        买入 = self.交易记录 [i]
        卖出 = self.交易记录 [i + 1]
        盈亏 = (卖出 ['价格'] - 买入 ['价格']) * 买入 ['数量']
        总盈亏 += 盈亏
        if 盈亏 > 0:
            盈利交易 += 1
    
    胜率 = 盈利交易 / 交易次数 if 交易次数 > 0 else 0
    
    # 盈亏比
    平均盈利 = 总盈亏 / 盈利交易 if 盈利交易 > 0 else 0
    平均亏损 = abs(总盈亏 / (交易次数 - 盈利交易)) if (交易次数 - 盈利交易) > 0 else 0
    盈亏比 = 平均盈利 / 平均亏损 if 平均亏损 > 0 else 0
    
    return {
        '交易次数': 交易次数，
        '胜率': 胜率，
        '盈亏比': 盈亏比，
        '总盈亏': 总盈亏
    }
```
