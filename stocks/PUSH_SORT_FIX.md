## 📋 排序代码

```python
def format_final_decision_message(stocks: list) -> str:
    # 按推荐程度排序
    rating_order = {'强烈推荐': 0, '推荐': 1, '谨慎推荐': 2, '关注': 3, '观望': 4}
    sorted_stocks = sorted(stocks, key=lambda x: (
        rating_order.get(x.get('rating', '观望'), 4),
        -x.get('confidence', 0),  # 同评级内按置信度降序
        -x.get('final_score', 0)  # 同置信度内按评分降序
    ))
    
    # 按评级分组显示
    current_rating = None
    for s in sorted_stocks[:10]:
        rating = s.get('rating', '')
        
        # 添加评级分组标题
        if rating != current_rating:
            current_rating = rating
            if rating == '强烈推荐':
                lines.append("━━━ ⭐⭐⭐ 强烈推荐 ━━━")
            elif rating == '推荐':
                lines.append("━━━ ⭐⭐ 推荐 ━━━")
            elif rating == '谨慎推荐':
                lines.append("━━━ ⭐ 谨慎推荐 ━━━")
            else:
                lines.append("━━━ ⭕ 观望 ━━━")
            lines.append("")
        
        # 显示股票信息
        ...
```
