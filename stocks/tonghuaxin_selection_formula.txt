{每周选股策略-选股条件}

RSI_N:=14;
RSI_LOW:=30;
RSI_HIGH:=50;

LC:=REF(C,1);
RSI_UP:=SMA(MAX(C-LC,0),RSI_N,1);
RSI_DOWN:=SMA(ABS(C-LC),RSI_N,1);
RSI_VALUE:=100*RSI_UP/(RSI_UP+RSI_DOWN);

MA5:=MA(C,5);
MA10:=MA(C,10);

RSI_OK:=RSI_VALUE>RSI_LOW AND RSI_VALUE<RSI_HIGH;
MA_OK:=MA5>MA10;
PRICE_OK:=C>MA5;

XG:RSI_OK AND MA_OK AND PRICE_OK;
